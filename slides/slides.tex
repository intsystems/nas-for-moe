\begin{frame}{План работы}
    \begin{block}{Цель исследования}
        Предложить метод поиска архитектуры для смеси экспертов.
    \end{block}
    \begin{block}{Проблема}
        При изменении архитектуры эксперта меняется результат роутинга, то есть меняются данные, на которых обучается эксперт.
    \end{block}
    \begin{block}{Предлагаемый метод}
        Предлагается оценивать качество архитектур каждого из экспертов с помощью суррогатной функции.
        Суррогатная функция обучается одновременно с оптимизацией назначений экспертов и поиска архитектур.
    \end{block}
\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Схема маршрутизации данных}
    Каждый элемент данных маршрутизируется одному из экспертов; архитектуры экспертов подбираются под назначенные им кластеры.
    \begin{figure}
        \centering
        \includegraphics[width=0.9\textwidth]{routing}
    \end{figure}

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Пространство поиска архитектуры смеси экспертов}

    \begin{block}{Архитектуры экспертов}
        Каждый эксперт --- свёрточная нейронная сеть, она задаётся ориентированным ациклическим графом, где каждый узел применяет операцию из множества $\mathcal{O}$. $\boldsymbol{\alpha}$ --- вектор выборов операций, по которому восстанавливается архитектура. Смесь из $K$ экспертов задаётся набором $\boldsymbol{\alpha}_1, \dots, \boldsymbol{\alpha}_K$.
    \end{block}

    \begin{block}{Кластеры данных}
        Данные разбиты на $M$ кластеров: $\mathcal{D} = \mathcal{C}_1 \sqcup \mathcal{C}_2 \sqcup \dots \sqcup \mathcal{C}_M$, $|\mathcal{C}_m|$ --- размер кластера $m$, а $m(n)$ --- номер кластера объекта $n$.
    \end{block}

    \begin{block}{Маршрутизация вектором $\mathbf{r}$}
        $r_{mk} \in [0,1]$ --- вероятность назначения кластера $m$ эксперту $k$, $\sum_{k=1}^{K} r_{mk} = 1$. Они образуют матрицу маршрутизации $\mathbf{R} \in [0,1]^{M \times K}$; её столбец $\mathbf{r}_k = \mathbf{R}_{:,k}$ задаёт кластеры эксперта $k$: $I_k = \{\, m : \arg\max_{k'} r_{mk'} = k \,\}$.
    \end{block}

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Суррогатная модель для оценки качества экспертов}

    Суррогатная функция $u(\boldsymbol{\alpha}, \mathbf{r})$ предсказывает среднюю по объектам кросс-энтропийную ошибку (loss) архитектуры $\boldsymbol{\alpha}$, обученной на данных, определяемых вектором $\mathbf{r}$:
    $$u: \mathcal{A} \times \{0,1\}^M \to \mathbb{R}_{\geq 0},$$
    где $\mathcal{A}$ -- пространство архитектур, $M$ -- число кластеров.

    \medskip

    Суррогат выражает пообъектное правдоподобие эксперта:
    $$u(\boldsymbol{\alpha}_k, \mathbf{r}_k) \approx -\log p\bigl(y_n \mid x_n, \boldsymbol{\alpha}_k, \boldsymbol{\theta}_k,\ m(n) \in I_k\bigr),$$
    где $\boldsymbol{\theta}_k$ --- параметры (веса) эксперта $k$.

    \bigskip

    Суррогат представляет собой графовую нейронную сеть, обучаемую на триплетах $(\boldsymbol{\alpha}, \mathbf{b}, \ell)$, где $\mathbf{b} \in \{0,1\}^M$ -- маска подмножества кластеров, $\ell$ -- кросс-энтропийная ошибка архитектуры, обученной на данных, соответствующих $\mathbf{b}$.

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Постановка задачи}
    \begin{block}{Максимизация правдоподобия смеси экспертов}
        \vspace{-8mm}
        \begin{gather*}
            \prod_{n=1}^{N} \sum_{k=1}^{K} p(y_n, m(n) \in I_k \mid x_n, \boldsymbol{\alpha}_k, \boldsymbol{\theta}_k) \to \max_{\boldsymbol{\alpha}_1, \dots, \boldsymbol{\alpha}_K, \boldsymbol{\theta}_1, \dots, \boldsymbol{\theta}_K},
        \end{gather*}
        где $K$ -- кол-во экспертов, $N$ -- кол-во элементов данных, $m(n)$ -- номер кластера элемента $n$. Для $k$-ого эксперта $\boldsymbol{\alpha}_k$ -- архитектура.
    \end{block}

    \begin{block}{Задача поиска архитектуры смеси из $K$ экспертов}
    \vspace{-5mm}
    \begin{gather*}
        \sum_{m=1}^{M} |\mathcal{C}_m|\, \log \sum_{k=1}^{K} r_{mk}\, e^{-u(\boldsymbol{\alpha}_k, \mathbf{r}_k)} \to \max_{\boldsymbol{\alpha}_1,\ldots,\boldsymbol{\alpha}_K,\, \mathbf{r}_1,\ldots,\mathbf{r}_K},
    \end{gather*}
    где $M$ --- число кластеров данных, $|\mathcal{C}_m|$ --- размер кластера $m$, $r_{mk}$ --- вероятность назначения кластера $m$ эксперту $k$, $\mathbf{r}_k = \{r_{mk}\}_{m=1}^{M}$.
    \end{block}

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Предлагаемый метод}

 Обобщенный EM-алгоритм с использованием суррогатной функции.

\begin{figure}[h]
  \begin{minipage}{1.\linewidth}
    \begin{algorithm}[H]
    \caption{SGEM}\label{alg:em}
        \begin{algorithmic}
            \State \textbf{Вход:} $K$, данные, кластеризация, $u^{(0)}$ --- начальный суррогат
            \For{$t = 0, 1, 2, \ldots$}
                \State \textbf{E-шаг:} $q_{mk}^{(t)} \propto r_{mk}^{(t)}\, e^{-u^{(t)}(\boldsymbol{\alpha}_k^{(t)}, \mathbf{r}_k^{(t)})}$
                \State \textbf{M-шаг:} $\sum_m |\mathcal{C}_m| \sum_k q^{(t)}_{mk} \left[\log r_{mk} - u^{(t)}(\boldsymbol{\alpha}_k, \mathbf{r}_k)\right] \to \max_{\boldsymbol{\alpha},\, \mathbf{R}}$
                \State \textbf{S-шаг:} добавить актуальные данные, переобучить $u$
            \EndFor
        \end{algorithmic}
    \end{algorithm}
  \end{minipage}
\end{figure}

\bigskip

На шаг $S$ суррогат $u^{(t)}$ дообучается и с ростом $t$ стремится к истинной оценке правдоподобия экспертов $u_{\mathrm{true}}$.

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Сходимость GEM с суррогатом}

    $L_{\mathrm{true}}$ --- функционал из постановки задачи с истинной оценкой правдоподобия $u_{\mathrm{true}}$ вместо суррогата.

    \begin{tcolorbox}[
    colback=gray!10, colframe=gray!50, arc=5mm,
    boxsep=0.5mm, left=3mm, right=3mm, top=2mm, bottom=2mm
    ]
    \textbf{Теорема (Бабкин, 2026).} Пусть $\{\boldsymbol{\psi}^{(t)} = (\boldsymbol{\alpha}^{(t)}, \mathbf{R}^{(t)})\}$ --- последовательность, порождаемая алгоритмом SGEM с суррогатами $u^{(t)}$. При достаточной скорости сходимости функционального ряда суррогатных функций к истинной оценке $u_{\mathrm{true}}$ и некоторых условиях регулярности выполняется:

    \medskip
    \textbf{(a)} последовательность истинных правдоподобий $\{L_{\mathrm{true}}(\boldsymbol{\psi}^{(t)})\}$ сходится;

    \medskip
    \textbf{(b)} любая предельная точка $\boldsymbol{\psi}^{\star}$ --- стационарная точка $L_{\mathrm{true}}$.
    \end{tcolorbox}

    \medskip

\textbf{Замечание:} если ошибка суррогата убывает экспоненциально с показателем $p > 1$, то $L_{\mathrm{true}}(\boldsymbol{\psi}^{(T)}) \to L_{\mathrm{true}}(\boldsymbol{\psi}^{\star})$ со скоростью $O(T^{1-p})$.

\end{frame}
