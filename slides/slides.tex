\begin{frame}{План работы}
    \begin{block}{Цель исследования:}
        Предложить метод поиска архитектуры для смеси экспертов.
    \end{block}
    \begin{block}{Проблема:}
        При изменении архитектуры эксперта меняется результат роутинга, то есть меняются данные, на которых обучается эксперт.
    \end{block}
    \begin{block}{Предлагаемый метод:}
        Предлагается оценивать качество архитектур каждого из экспертов с помощью суррогатной функции.
        Суррогатная функция обучается одновременно с оптимизацией назначений экспертов и поиска архитектур.
    \end{block}
\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Схема работы}
    Архитектуры экспертов и кластеры данных подбираются для каждого кластера с помощью суррогатной модели $u$.

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Суррогатная модель для оценки качества экспертов}

    Суррогатная функция $u(\alpha, R)$ предсказывает среднюю по объектам кросс-энтропийную ошибку (loss) архитектуры $\alpha$, обученной на данных, определяемых вектором $R$:
    $$u: \mathcal{A} \times \{0,1\}^M \to \mathbb{R}_{\geq 0},$$
    где $\mathcal{A}$ -- пространство архитектур, $M$ -- число кластеров.

    \medskip

    Суррогат выражает пообъектное правдоподобие эксперта (отрицательное лог-правдоподобие):
    $$u(\alpha_k, R_k) \approx -\log p\bigl(y_n \mid x_n, \alpha_k, \theta_k,\ m(n) \in C_k\bigr).$$

    \bigskip

    Суррогат представляет собой графовую нейронную сеть, обучаемую на триплетах $(\alpha, b, \ell)$, где $b \in \{0,1\}^M$ -- маска подмножества кластеров, $\ell$ -- кросс-энтропийная ошибка архитектуры, обученной на данных, соответствующих $b$.

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Постановка задачи}
    \begin{block}{Максимизация правдоподобия смеси экспертов}
        \vspace{-8mm}
        \begin{gather*}
            \prod_{n=1}^{N} \sum_{k=1}^{K} p(y_n, m(n) \in C_k \mid x_n, \alpha_k, \theta_k) \to \max_{\alpha_1, \dots, \alpha_K, \theta_1, \dots, \theta_K},
        \end{gather*}
        где $K$ -- кол-во экспертов, $N$ -- кол-во элементов данных, $m(n)$ -- номер кластера элемента $n$. Для $k$-ого эксперта $C_k$ -- множество индексов назначенных ему кластеров, $\theta_k$ -- параметры, $\alpha_k$ -- архитектура.
    \end{block}

    \begin{block}{Задача поиска архитектуры смеси из $K$ экспертов}
    \vspace{-5mm}
    \begin{gather*}
        \sum_{m=1}^{M} |\mathcal{C}_m|\, \log \sum_{k=1}^{K} r_{mk}\, e^{-u(\alpha_k, R_k)} \to \max_{\alpha_1,\ldots,\alpha_K,\, R_1,\ldots,R_K},
    \end{gather*}
    где $M$ --- число кластеров данных, $|\mathcal{C}_m|$ --- размер кластера $m$, $r_{mk}$ --- вероятность назначения кластера $m$ эксперту $k$, $R_k = \{r_{mk}\}_{m=1}^{M}$.
    \end{block}

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Предлагаемый метод}

 Обобщенный EM-алгоритм с использованием суррогатной функции.

\begin{figure}[h]
  \begin{minipage}{1.\linewidth}
    \begin{algorithm}[H]
    \caption{SGEM}\label{alg:em}
        \begin{algorithmic}
            \State \textbf{Вход:} $K$, данные, кластеризация, $u^{(0)}$ --- начальный суррогат
            \For{$t = 0, 1, 2, \ldots$}
                \State \textbf{E-шаг:} $q_{mk}^{(t)} \propto r_{mk}^{(t)}\, e^{-u^{(t)}(\alpha_k^{(t)}, R_k^{(t)})}$
                \State \textbf{M-шаг:} $\sum_m |\mathcal{C}_m| \sum_k q^{(t)}_{mk} \left[\log r_{mk} - u^{(t)}(\alpha_k, R_k)\right] \to \max_{\overline{\alpha}, \overline{R}}$
                \State \textbf{S-шаг:} добавить актуальные данные, переобучить $u$
            \EndFor
        \end{algorithmic}
    \end{algorithm}
  \end{minipage}
\end{figure}

\bigskip

На шаг $S$ суррогат $u^{(t)}$ дообучается и с ростом $t$ стремится к истинной оценке правдоподобия экспертов $u_{\mathrm{true}}$.

\end{frame}
%----------------------------------------------------------------------------------------------------------
\begin{frame}{Сходимость GEM с адаптивным суррогатом}
    \begin{tcolorbox}[
    colback=gray!10, colframe=gray!50, arc=5mm,
    boxsep=0.5mm, left=3mm, right=3mm, top=2mm, bottom=2mm
    ]
    \textbf{Теорема.} Пусть $\{\theta^{(t)} = (\alpha^{(t)}, r^{(t)})\}_{t=1}^\infty$ --- последовательность GEM-алгоритма с суррогатом $u^{(t)}$, и выполнены условия: (1) равномерная ограниченность $u_{true}, \{u^{(t)}\}_{t=1}^\infty$, (2) $\sum_{t=1}^\infty \varepsilon_t < \infty$ где $\varepsilon_t = \sup |u^{(t)} - u_{\mathrm{true}}|$, (3) на $M$-шаге $F$-функция не убывает, (4) $u(\alpha, r)$ непрерывно по второму аргументу $\forall \alpha \in \mathcal{A}$. Тогда:

    \medskip
    \textbf{(a)} $\{L_{\mathrm{true}}(\theta^{(t)})\}$ сходится.

    \textbf{(b)} Любая предельная точка $\theta^{\star}$ является стационарной:
    $Q_{\mathrm{true}}(\theta;\, \theta^{\star}) \leq Q_{\mathrm{true}}(\theta^{\star};\, \theta^{\star})$ для всех $\theta \in \Theta$, где $Q_{\mathrm{true}}(\theta;\, \theta') = \sum_m |\mathcal{C}_m| \sum_k w_{mk}(\theta')\, \bigl[\log r_{mk} - u_{\mathrm{true}}(\alpha_k, R_k)\bigr]$ и $w_{mk}(\theta') = q_{mk}^{\star}(\theta')$.
    \end{tcolorbox}

    \medskip

\textbf{Замечание:} если $\varepsilon_t$ убывает экспоненциально с показателем $p > 1$, то $L_{\mathrm{true}}(\theta^{(T)}) \to L_{\mathrm{true}}(\theta^{\star})$ со скоростью $O(T^{1-p})$.    

\end{frame}
